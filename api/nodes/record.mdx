## Overview

The Record node represents a single row in a database. Each record contains field values corresponding to the database's field definitions, and can optionally have a rich text document attached.

## Attributes Schema

```typescript
z.object({
  type: z.literal('record'),
  parentId: z.string(),
  databaseId: z.string(),
  name: z.string(),
  avatar: z.string().nullable().optional(),
  fields: z.record(z.string(), fieldValueSchema),
})
```

<ParamField path="type" type="literal" required>
  Must be `'record'`
</ParamField>

<ParamField path="parentId" type="string" required>
  ID of the parent database node
</ParamField>

<ParamField path="databaseId" type="string" required>
  ID of the database this record belongs to (same as parentId)
</ParamField>

<ParamField path="name" type="string" required>
  The name/title of the record (displayed in the Name column)
</ParamField>

<ParamField path="avatar" type="string | null">
  Optional avatar URL or emoji for the record
</ParamField>

<ParamField path="fields" type="Record<string, FieldValue>" required>
  Map of field IDs to their values in this record. Field IDs must match the database's field definitions.
  
  <Expandable title="Field Value Types">
    Each value must conform to one of these schemas:
    
    ```typescript
    // Boolean value
    { type: 'boolean', value: boolean }
    
    // String value (for email, phone, url, etc.)
    { type: 'string', value: string }
    
    // String array (for multi-select, collaborators, etc.)
    { type: 'string_array', value: string[] }
    
    // Number value
    { type: 'number', value: number }
    
    // Rich text value
    { type: 'text', value: Text }
    ```
  </Expandable>
</ParamField>
